// Example program
#include <iostream>
using namespace std;
int main()
{
    int numOfProb;
    cin >> numOfProb;
    int numOfProbSolved = 0 ;
    
    for( int i=0; i < numOfProb; ++i ){  
        bool mem1, mem2, mem3;
    
        cin >> mem1 >> mem2 >> mem3;
        if( mem1 + mem2 + mem3 >= 2 ) {
            numOfProbSolved += 1;
           
            }
        else {
            numOfProbSolved +=0;
            }
            
        }
    cout << numOfProbSolved ;
}
